  def load(str=nil)
    case language
    when :boolexp
      self.src = str
      log.info "Load of BoolExp program - #{name}"
      true
    when :fart
      File.open("farts/#{name}.fart") {|f|
        self.src = f.read
      }
      log.info "Load of FART program - #{name}"
      true
    else
      false
    end
  end
