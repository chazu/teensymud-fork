  def compile
    case language
    when :boolexp
      true
    when :fart
      @prog = Farts::Parser.new.parse(src)
      log.info "Compile of FART program - #{name}"
      true
    else
      false
    end
  rescue Exception
    log.error $!
    @prog = nil
    false
  end
