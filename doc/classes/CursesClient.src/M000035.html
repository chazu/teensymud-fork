  def update(msg)
    case msg
    when Connection, :initdone
      super(msg)
    when :disconnected
      unsubscribe_all
      $shutdown = true
      Curses.addstr "Disconnected."
      exit
    when String
      Curses.addstr(msg)
      Curses.refresh
    else
      Curses.addstr "Unknown msg - #{msg.inspect}"
    end
  end
