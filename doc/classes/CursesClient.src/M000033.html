  def initialize(opts)
    super(opts)
    Curses.init_screen
    Curses.cbreak
    Curses.noecho if @opts.echo
    Curses.nl
    Curses.stdscr.scrollok(true)
    if !$opts.win32  # Need workaround for these
      Curses.stdscr.keypad(true)
      Curses.timeout = 0
    end
    Curses.start_color
  end
