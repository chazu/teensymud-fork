  def initialize(db)
    @cwidth = options['cache_width']
    @cdepth = options['cache_depth']

    @db =  db
    @st = CacheStats.new

    @cache = Array.new(@cwidth) {Array.new}
    @cwidth.times do |i|
      @cdepth.times do |j|
        @cache[i] << CacheEntry.new
        @st.inc(:cache_size)
      end
    end

  end
