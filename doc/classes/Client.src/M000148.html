  def update(msg)
    case msg
    when Connection
      @conn = msg
      unsubscribe_all
      msg.subscribe(self)
      self.subscribe(msg)
      if @opts.win32
        @conn.set(:terminal, "vt100")
      else
        @conn.set(:terminal, "xterm")
      end
    when :initdone
      @conn.set(:termsize, [80,43])
    end
  end
