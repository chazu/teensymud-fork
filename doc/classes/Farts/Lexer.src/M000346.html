  def next_token
    @token = [:UNKNOWN, nil]
    scan(/\s+/)
    check_unary if @token[0] == :UNKNOWN
    check_binary if @token[0] == :UNKNOWN
    check_string if @token[0] == :UNKNOWN
    check_numeric if @token[0] == :UNKNOWN
    check_ident if @token[0] == :UNKNOWN
    if @token[0] == :COMMENT
      @token[1] += skip_line
    end
    if @token[0] == :UNKNOWN
      @token[1] = scan_until(/\s+/)
    end
    return [false,false] if eos?
    return @token
  end
