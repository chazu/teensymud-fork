  def cmd_go(args)
    case args
    when nil, ""
      sendto("Where do you want to go?")
    else
      ex = get_object(location).exits.keys.grep(/^#{args}/)
      if ex.empty?
        sendto("Can't find that place")
      elsif ex.size > 1
        ln = "Which did you mean, "
        ex.each {|x| ln += "\'" + x + "\' "}
        ln += "?"
        sendto(ln)
      else
        add_event(id,location,:leave,ex[0])
      end
    end
  end
