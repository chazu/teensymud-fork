  def cmd_object(args)
    case args
    when /(.*)/
      newobj = GameObject.new($1, id, location)
      if newobj.nil?
        log.error "Unable to create object."
        sendto "System error: unable to create object."
        return
      end
      put_object(newobj)
      get_object(location).add_contents(newobj.id)
      sendto "Ok."
    else
      sendto("What!!?")
    end
  end
