  def cmd_help(args)
    case args
    when nil, ""
      sendto("====================HELP=======================")
      world.cmds.to_hash.values.each do |h|
        sendto(sprintf("%-10s - %s", h.name, h.help))
      end
    else
      q = world.cmds.find(args)
      if q
        sendto("====================HELP=======================")
        q.each do |h|
          sendto(sprintf("%-10s - %s", h.name, h.help))
        end
      else
        sendto "No help on that."
      end
    end
  end
